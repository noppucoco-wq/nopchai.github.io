    <script>
        // Function to handle Navigation
        function navigate(viewId) {
            // 1. Hide all views
            document.querySelectorAll('.page-view').forEach(view => {
                view.classList.remove('active');
            });
            
            // 2. Show the selected view
            document.getElementById('view-' + viewId).classList.add('active');

            // 3. Update styling for Desktop Navigation links
            document.querySelectorAll('.nav-btn').forEach(btn => {
                btn.classList.remove('text-brand-800', 'border-b-2', 'border-brand-800');
                btn.classList.add('text-slate-600');
                
                if(btn.dataset.target === viewId) {
                    btn.classList.add('text-brand-800', 'border-b-2', 'border-brand-800');
                    btn.classList.remove('text-slate-600');
                }
            });

            // 4. Update styling for Mobile Navigation links
            document.querySelectorAll('.mobile-nav-btn').forEach(btn => {
                btn.classList.remove('text-brand-800', 'font-medium');
                btn.classList.add('text-slate-600');
                
                if(btn.dataset.target === viewId) {
                    btn.classList.add('text-brand-800', 'font-medium');
                    btn.classList.remove('text-slate-600');
                }
            });

            // 5. Close mobile menu if it's open
            document.getElementById('mobile-menu').classList.add('hidden');

            // 6. Scroll smoothly to the top of the page
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        // Function to toggle Mobile Menu open/close
        function toggleMobileMenu() {
            document.getElementById('mobile-menu').classList.toggle('hidden');
        }

        // Initialize: Show home page on first load
        window.onload = function() {
            navigate('home');
        };
    </script>
